 60 /**
 61  * You probably want to just use the global variable <code>traceManager</code>.
 62  * 
 63  * @class The TraceManager manages a sort of stack frame that is memoized on
 64  *        disk (using the JavaScript database associated with the current file).
 65  *        All stack frames are memoized, so the result is a sort of stack tree.
 66  * 
 67  * <p>
 68  * <code>traceManager</code> is a global instance of this class.
 69  * </p>
 70  */
 71 function TraceManager() {
 72 	this.stackFramePath = ["__stackFrames", "for-real"]
 73 	this.stackIndexes = [0]
 74 	this.visitedStackFrames = {}
 75 	this.beforePushFrame = true
 76 }
 77 
 78 /**
 79  * Pushes a new stack frame onto the memoized stack.
 80  */
 81 TraceManager.prototype.pushFrame = function(frameName) {
 82 	this.beforePushFrame = false
 83 	if (frameName) {
 84 		this.stackFramePath.push("namedFrames")
 85 		this.stackFramePath.push(frameName)
 86 	} else {
 87 		this.stackFramePath.push("sequencialFrames")
 88 		this.stackFramePath
 89 				.push(this.stackIndexes[this.stackIndexes.length - 1]++)
 90 	}
 91 	this.stackIndexes.push(0)
 92 
 93 	var path = json(this.stackFramePath)
 94 
 95 	// make sure we haven't already been here
 96 	if (this.visitedStackFrames[path]) {
 97 		throw "visiting the same stack frame twice: " + path
 98 	}
 99 	this.visitedStackFrames[path] = true
100 
101 	return database.query("return prune(ensure(null, " + path + ", " + json({
102 				creationTime : time()
103 			}) + "))")
104 }
105 
106 /**
107  * Sets the value of a "local variable" in the current memoized stack frame.
108  * This value is persisted in the memoized stack.
109  */
110 TraceManager.prototype.setFrameValue = function(name, value) {
111 	database.query("ensure(null, " + json(this.stackFramePath.concat(["values"]))
112 			+ ")[" + json(name) + "] = " + json(value))
113 }
114 
115 /**
116  * Gets the value of a "local variable" in the current memoized stack frame.
117  */
118 TraceManager.prototype.getFrameValue = function(name) {
119 	database.query("return ensure(null, "
120 			+ json(this.stackFramePath.concat(["values"])) + ")[" + json(name)
121 			+ "]")
122 }

124 /**
125  * Pops the top stack frame off of the memoized stack. Note that this does not
126  * remove anything from the memoized stack. The memoized stack is a tree which
127  * stores a trace of every stack frame that ever went onto the stack.
128  */
129 TraceManager.prototype.popFrame = function() {
130 	this.stackFramePath.pop()
131 	this.stackFramePath.pop()
132 	this.stackIndexes.pop()
133 }
134 
135 /**
136  * Calls the function <i>func</i> only once; subsequent runs of the program
137  * will not call <i>func</i> if it returns successfully this time. If <i>func</i>
138  * throws an exception, then it will be called again when the program is
139  * re-executed.
140  * 
141  * <p>
142  * This function creates a new memoized stack frame before calling <i>func</i>.
143  * You can give this stack frame a name as an optional second parameter
144  * <i>frameName</i>.
145  * </p>
146  */
147 TraceManager.prototype.once = function(func, frameName) {
148 	var frame = this.pushFrame(frameName)
149 	try {
150 		if ("onceFunc" in frame) {
151 			if (frame.onceFunc != "" + func) {
152 				throw "The memoized stack is inconsistent with the program. The program is running: " + func + "\nBut on a previous run, it ran: " + frame.onceFunc 
153 			}
154 		} else {
155 			database.query("merge(ensure(null, " + json(this.stackFramePath)
156 					+ "), " + json({onceFunc : "" + func}) + ")")
157 		}
158 	
159 		if ("returnValue" in frame) {
160 			frame.returnValue = database.query("return ensure(null, "
161 					+ json(this.stackFramePath) + ").returnValue")
162 			if ("printOutput" in frame) {
163 				Packages.java.lang.System.out["print(java.lang.String)"](frame.printOutput)
164 			}
165 		} else {
166 			var wireTap = new Packages.edu.mit.csail.uid.turkit.util.WireTap()
167 			var returnValue = func()
168 			var printOutput = wireTap.close()
169 			var returnTime = time()		
170 			frame = {
171 				returnValue : returnValue,
172 				returnTime : returnTime,
173 				printOutput : printOutput
174 			}
175 			database.query("merge(ensure(null, " + json(this.stackFramePath)
176 					+ "), " + json(frame) + ")")
177 		}
178 	} finally {
179 		this.popFrame()
180 	}
181 	return frame.returnValue
182 }

185  * This is a wrapper around {@link TraceManager#once}.

191 /**
192  * Calls the function <i>func</i> in a new memoized stack frame, and catches
193  * the "stop" exception. Returns <code>true</code> if the call succeeds.
194  * Returns <code>false</code> if <i>func</i> throws a "stop" exception (see
195  * {@link TraceManager#stop}).
196  */
197 TraceManager.prototype.attempt = function(func, frameName) {
198 	this.pushFrame(frameName)
199 	try {
200 		func()
201 	} catch (e) {
202 		if ("" + e == "stop") {
203 			return false
204 		} else {
205 			throw e
206 		}
207 	} finally {
208 		this.popFrame()
209 	}
210 	return true
211 }

256 			+ json(this.stackFramePath[1]) + ") {delete a[k]}})")

311 /*
312  * Throw an exception if we cannot spend the given amount of <i>money</i>, or
313  * create the given number of <i>hits</i>, without violating our safety limits.
314  * The <i>callbackBeforeCrash</i> will be called before an exception is thrown,
315  * if an exception is about to be thrown.
316  */
317 MTurk.prototype.assertWeCanSpend = function(money, hits, callbackBeforeCrash) {
318 	if (!javaTurKit.safety)
319 		return
320 
321 	var safety = {
322 		moneySpent : 0,
323 		hitsCreated : 0
324 	}
325 	safety = database.query("return ensure('__safetyCounters', " + json(safety)
326 			+ ")")
327 	safety.moneySpent += money
328 	safety.hitsCreated += hits
329 	if (safety.moneySpent > javaTurKit.maxMoney) {
330 		if (callbackBeforeCrash)
331 			callbackBeforeCrash()
332 		throw "TurKit has detected a safety violation: spending too much money."
333 				+ "You need to increase your spending limit with TurKit (not with MTurk) to overcome this problem."
334 	}
335 	if (safety.hitsCreated > javaTurKit.maxHITs) {
336 		if (callbackBeforeCrash)
337 			callbackBeforeCrash()
338 		throw "TurKit has detected a safety violation: creating too many HITs."
339 				+ "You need to increase your hit limit with TurKit (not with MTurk) to overcome this problem."
340 	}
341 	safety = database.query("__safetyCounters = " + json(safety))
342 }

344 /**
345  * Repeatedly tries to execute the given <i>func</i>, and returns the result
346  * once it succeeds.
347  */
348 MTurk.prototype.keepTrying = function(func) {
349 	sleep(0.05)
350 	var waitTime = 0.1
351 	for (var i = 0; i < 10; i++) {
352 		try {
353 			var ret = func()
354 			return ret
355 		} catch (e) {
356 			if (/throttled/.exec("" + e)) {
357 				verbosePrint("throttled")
358 				sleep(waitTime)
359 				waitTime += 0.1
360 			} else {
361 				throw e
362 			}
363 		}
364 	}
365 	// try one last time (this time don't catch the error)
366 	return func()
367 }

378 /**

380  * <ul>

382  * <li><b>description</b>: <b>desc</b> is also accepted. A slightly longer
383  * description of the HIT, also shown in the list of HITs on MTurk</li>
384  * <li><b>question</b>: a string of XML specifying what will be shown. <a
385  * href="http://docs.amazonwebservices.com/AWSMechanicalTurkRequester/2008-08-02/index.html?ApiReference_QuestionFormDataStructureArticle.html">See
386  * documentation here</a>. Instead of <i>question</i>, you may use the following special parameters:
387  <ul>
388  <li><b>url</b>: creates an external question pointing to this URL</li>
389  <li><b>height</b>: (optional) height of the iFrame embedded in MTurk, in pixels (default is 600).</li>

400  * <li><b>keywords</b>: keywords to help people search for your HIT.</li>

406  * <li><b>maxAssignments</b>: default is 1.</li>

409  * <li><b>minApproval</b>: minimum approval percentage. The appropriate
410  * requirement will be added if you supply a percentage here.</li>
411  * <li><b>responseGroup</b>: default is no response group.</li>
412  * </ul>
413  */
414 MTurk.prototype.createHITRaw = function(params) {
415 	if (!params)
416 		params = {}
417 
418 	if (params.HITTypeID)
419 		params.hitTypeId = params.HITTypeID
420 	if (params.hitTypeId) {
421 	} else {
422 		params.hitTypeId = null
423 		if (!params.title)
424 			throw "createHIT requires a title"
425 
426 		if (params.desc)
427 			params.description = params.desc
428 		if (!params.description)
429 			throw "createHIT requires a description"
430 
431 		if (!params.reward)
432 			throw "createHIT requires a reward"
433 
434 		if (!params.assignmentDurationInSeconds)
435 			params.assignmentDurationInSeconds = 60 * 60 // one hour
436 
437 		if (params.minApproval) {
438 			ensure(params, "qualificationRequirements", [])
439 					.push(new Packages.com.amazonaws.mturk.requester.QualificationRequirement(
440 							"000000000000000000L0",
441 							Packages.com.amazonaws.mturk.requester.Comparator.GreaterThanOrEqualTo,
442 							params.minApproval, null, false))
443 		}
444 	}
445 
446 	if (params.html) {
447 		if (!params.bucket) {
448 			params.bucket = javaTurKit.awsAccessKeyID.toLowerCase() + "-turkit"
449 		}
450 		var s = ("" + javaTurKit.taskTemplate).replace(/\[\[\[CONTENT\]\]\]/, params.html)
451 		var key = Packages.edu.mit.csail.uid.turkit.util.U.md5(s) + ".html"
452 		params.url = s3.putString(params.bucket, key, s)
453 		
454 		if (params.blockWorkers) {
455 			if ((typeof params.blockWorkers) != "string") {
456 				params.blockWorkers = params.blockWorkers.join(",")
457 			}
458 			params.url += "?blockWorkers=" + params.blockWorkers
459 		}
460 	}
461 	
462 	if (params.url) {
463 		if (!params.height)
464 			params.height = 600
465 
466 		params.question = '<ExternalQuestion xmlns="http://mechanicalturk.amazonaws.com/AWSMechanicalTurkDataSchemas/2006-07-14/ExternalQuestion.xsd">'
467 				+ '<ExternalURL>'
468 				+ escapeXml(params.url)
469 				+ '</ExternalURL>'
470 				+ '<FrameHeight>'
471 				+ params.height
472 				+ '</FrameHeight>'
473 				+ '</ExternalQuestion>'
474 	}
475 
476 	if (!params.question)
477 		throw "createHIT requires a question (or a url, or html)"
478 
479 	if (!params.lifetimeInSeconds)
480 		params.lifetimeInSeconds = 60 * 60 * 24 * 7 // one week
481 
482 	if (!params.maxAssignments)
483 		params.maxAssignments = 1
484 
485 	if (!params.autoApprovalDelayInSeconds)
486 		params.autoApprovalDelayInSeconds = null
487 
488 	if (javaTurKit.safety) {
489 		if (params.hitTypeId) {
490 			ensure(params, 'responseGroup', []).push('HITDetail')
491 		} else {
492 			this.assertWeCanSpend(params.reward * params.maxAssignments,
493 					params.maxAssignments)
494 		}
495 	}
496 
497 	if (!params.requesterAnnotation)
498 		params.requesterAnnotation = null
499 
500 	if (!params.responseGroup)
501 		params.responseGroup = null
502 
503 	if (!params.qualificationRequirements)
504 		params.qualificationRequirements = null
505 
506 	var hit = this.keepTrying(function() {
507 				return mturk.requesterService.createHIT(params.hitTypeId,
508 						params.title, params.description, params.keywords,
509 						params.question, params.reward,
510 						params.assignmentDurationInSeconds,
511 						params.autoApprovalDelayInSeconds,
512 						params.lifetimeInSeconds, params.maxAssignments,
513 						params.requesterAnnotation,
514 						params.qualificationRequirements, params.responseGroup);
515 			})
516 
517 	if (javaTurKit.safety) {
518 		if (params.hitTypeId) {
519 			this.assertWeCanSpend(parseFloat(hit.getReward().getAmount())
520 							* params.maxAssignments, params.maxAssignments,

523 					})
524 		} else {
525 		}
526 	}

531 					? "https://workersandbox.mturk.com/mturk/preview?groupId="
532 					: "https://www.mturk.com/mturk/preview?groupId=")
533 			+ hit.getHITTypeId()
534 	verbosePrint("        url: " + url)
535 	database.query("ensure(null, ['__HITs', " + json(javaTurKit.mode + ":" + hitId) + "], " + json({url : url}) + ")")
536 	return hitId
537 }

641 /**
642 	Deletes the given <code>hit</code>.
643 	If there are any completed assignments that have not been approved or rejected, then they are approved.
644  */
645 MTurk.prototype.deleteHITRaw = function(hit) {
646 	var hitId = this.tryToGetHITId(hit)
647 	
648 	// first, try to disable the HIT
649 	try {
650 		this.keepTrying(function() {
651 			mturk.requesterService.disableHIT(hitId)
652 		})
653 		verbosePrint("disabled HIT: " + hitId)
654 	} catch (e) {
655 		try {	
656 			// check to see if we have already deleted it
657 			hit = this.getHIT(hitId)
658 			if (hit.hitStatus == "Disposed") {
659 				verbosePrint("already deleted HIT: " + hitId)
660 			} else {
661 				// ok, it must be "Reviewable"
662 				// (since we couldn't disable it, and it isn't already deleted)
663 				
664 				// first, approve all the assignments
665 				foreach(hit.assignments, function (a) {
666 					if (a.assignmentStatus == "Submitted")
667 						mturk.approveAssignmentRaw(a)
668 				})
669 				
670 				// next, dispose of the HIT
671 				this.keepTrying(function() {
672 					mturk.requesterService.disposeHIT(hitId)
673 				})
674 				verbosePrint("disposed HIT: " + hitId)
675 			}
676 		} catch (e) {
677 			if (/AWS\.MechanicalTurk\.HITDoesNotExist/.exec("" + e)) {
678 				verbosePrint("HIT not found: " + hitId)
679 			} else {
680 				throw e
681 			}
682 		}
683 	}
684 	
685 	database.query("delete __HITs[" + json(javaTurKit.mode + ":" + hitId) + "]")
686 }

811 /**
812  * Returns an object representing all of the information MTurk has on the given
813  * HIT, including the assigments, and the associated answer data. The returned
814  * object will have a value called <i>done</i> set to true iff all the pending
815  * assigments for this HIT have been completed.
816  * 
817  * <p>
818  * Note that the <i>answer</i> data structure associated with each assigment is
819  * simplified. It is recommended that you print out the result of this function
820  * using {@link json}, in order to know what it looks like for your specific
821  * situation.
822  * </p>
823  */
824 MTurk.prototype.getHIT = function(hit) {
825 	var hitId = this.tryToGetHITId(hit)
826 	var hit = this.keepTrying(function() {
827 				return mturk.requesterService.getHIT(hitId)
828 			})
829 	var hit = {
830 		hitId : "" + hit.getHITId(),
831 		hitTypeId : "" + hit.getHITTypeId(),
832 		title : "" + hit.getTitle(),
833 		description : "" + hit.getDescription(),
834 		keywords : "" + hit.getKeywords(),
835 		reward : "" + hit.getReward().getAmount(),
836 		question : hit.getQuestion(),
837 		maxAssignments : "" + hit.getMaxAssignments(),
838 		assignmentDurationInSeconds : hit.getAssignmentDurationInSeconds(),
839 		autoApprovalDelayInSeconds : hit.getAutoApprovalDelayInSeconds(),
840 		requesterAnnotation : hit.getRequesterAnnotation(),
841 		hitStatus : "" + hit.getHITStatus(),
842 		hitReviewStatus : "" + hit.getHITReviewStatus(),
843 		creationTime : hit.getCreationTime().getTimeInMillis(),
844 		expiration : hit.getExpiration().getTimeInMillis(),
845 		assignments : []
846 	}
847 	foreach(this.getAssignmentsForHIT(hitId), function(javaAssignment) {
848 				function getTime(t) {
849 					if (t != null)
850 						return t.getTimeInMillis()
851 				}
852 				var assignment = {
853 					assignmentId : "" + javaAssignment.getAssignmentId(),
854 					hitId : "" + javaAssignment.getHITId(),
855 					assignmentStatus : ""
856 							+ javaAssignment.getAssignmentStatus(),
857 					requesterFeedback : ""
858 							+ javaAssignment.getRequesterFeedback(),
859 					workerId : "" + javaAssignment.getWorkerId(),
860 					acceptTime : getTime(javaAssignment.getAcceptTime()),
861 					approvalTime : getTime(javaAssignment.getApprovalTime()),
862 					autoApprovalTime : getTime(javaAssignment
863 							.getAutoApprovalTime()),
864 					deadline : getTime(javaAssignment.getDeadline()),
865 					rejectionTime : getTime(javaAssignment.getRejectionTime()),
866 					submitTime : getTime(javaAssignment.getSubmitTime())
867 				}
868 
869 				// deal with answers
870 				// NOTE: we use the name "answer" to refer to the set of
871 				// answers, because that is how Mechanical Turk refers to them
872 				// in the
873 				// Assignment data structure
874 				assignment.answer = {}
875 				var answers = mturk.requesterService.parseAnswers(javaAssignment
876 						.getAnswer()).getAnswer()
877 				for (var ai = 0; ai < answers.size(); ai++) {
878 					var a = answers.get(ai)
879 
880 					var rhs = a.getFreeText()
881 					if (rhs) {
882 						rhs = "" + rhs
883 					}
884 					if (!rhs) {
885 						rhs = a.getUploadedFileKey()
886 						if (rhs) {
887 							rhs = {
888 								uploadedFileKey : "" + rhs,
889 								uploadedFileSizeInBytes : a
890 										.getUploadedFileSizeInBytes()
891 							}
892 						}
893 					}
894 					if (!rhs) {
895 						rhs = []
896 						var sels = a.getSelectionIdentifier()
897 						for (var si = 0; si < sels.size(); si++) {
898 							rhs.push("" + sels.get(si))
899 						}
900 						var o = a.getOtherSelectionText()
901 						if (o) {
902 							rhs.push("" + o)
903 						}
904 					}
905 
906 					assignment.answer["" + a.getQuestionIdentifier()] = rhs
907 				}
908 
909 				hit.assignments.push(assignment)
910 			})
911 	hit.done = (hit.hitStatus == "Reviewable")
912 			&& (hit.assignments.length == hit.maxAssignments)
913 	return hit
914 }

936 /**
937  * Collects multiple votes until a minimum number of necessary votes for a
938  * single choice is achieved. The <code>hit</code> supplied to this function
939  * must have its <i>maxAssignments</i> set to the minimum number of votes
940  * necessary for a single choice. This function will add assignments if
941  * necessary. This function relies on <code>extractVoteFromAnswer</code> to
942  * extract a String representing the choice given an answer data structure (see
943  * {@link MTurk#getHIT}).
944  * 
945  * <p>
946  * The return value is an object with the following entries:
947  * </p>
948  * <ul>
949  * <li><b>bestOption</b>: the String representing the choice with the most
950  * votes.</li>
951  * <li><b>totalVoteCount</b>: the number of votes received from turkers.</li>
952  * <li><b>voteCounts</b>: an object with key/value pairs representing
953  * choice/voteCounts.</li>
954  * <li><b>hit</b>: the HIT data structure representing the HIT on the final iteration of the voting process.</li>
955  * </ul>
956  */
957 MTurk.prototype.vote = function(hit, extractVoteFromAnswer) {
958 	var necessaryVoteCount = null
959 	var hitId = this.tryToGetHITId(hit)
960 	while (true) {
961 		var hit = this.waitForHIT(hitId)
962 
963 		if (necessaryVoteCount == null) {
964 			necessaryVoteCount = hit.maxAssignments
965 		}
966 
967 		var votes = {}
968 		foreach(hit.assignments, function(assignment) {
969 					var vote = extractVoteFromAnswer(assignment.answer)
970 					votes[vote] = ensure(votes, [vote], 0) + 1
971 				})
972 		var winnerVotes, winner
973 		[winnerVotes, winner] = getMax(votes)
974 
975 		if (winnerVotes >= necessaryVoteCount) {
976 			foreach(hit.assignments, function(assignment) {
977 						mturk.approveAssignment(assignment)
978 					})
979 			this.deleteHIT(hit)
980 			return {
981 				bestOption : winner,
982 				totalVoteCount : hit.assignments.length,
983 				voteCounts : votes,
984 				hit : hit
985 			}
986 		} else {
987 			this.extendHIT(hit, necessaryVoteCount - winnerVotes, null)
988 		}
989 	}
990 }

992 /**
993  * Works just like the JavaScript array sort function, except that this one can
994  * perform comparisons in parallel on MTurk by catching the "stop" exception
995  * which is thrown when waiting on an MTurk HIT using {@link MTurk#waitForHIT}.
996  */
997 MTurk.prototype.sort = function(a, comparator) {
998 	traceManager.pushFrame()
999 	try {
1000 		var sortTree = traceManager.getFrameValue("sortTree")
1001 		if (!sortTree)
1002 			sortTree = {}
1003 
1004 		function insertIntoTree(index, tree) {
1005 			if (tree.index == null) {
1006 				tree.index = index
1007 				return
1008 			}
1009 			var comp = comparator(a[index], a[tree.index])
1010 			if (comp == 0) {
1011 			} else if (comp < 0) {
1012 				insertIntoTree(index, ensure(tree, ["left"], {}))
1013 			} else {
1014 				insertIntoTree(index, ensure(tree, ["right"], {}))
1015 			}
1016 		}
1017 		var done = true
1018 		foreach(a, function(e, i) {
1019 					if (attempt(function() {
1020 								insertIntoTree(i, sortTree)
1021 							})) {
1022 						traceManager.setFrameValue("sortTree", sortTree)
1023 					} else {
1024 						done = false
1025 					}
1026 				})
1027 		if (!done)
1028 			stop()
1029 
1030 		var newA = []
1031 		function traverseTree(tree) {
1032 			if (!tree)
1033 				return
1034 			traverseTree(tree.left)
1035 			newA.push(a[tree.index])
1036 			traverseTree(tree.right)
1037 		}
1038 		traverseTree(sortTree)
1039 		return newA
1040 	} finally {
1041 		traceManager.popFrame()
1042 	}
1043 }
