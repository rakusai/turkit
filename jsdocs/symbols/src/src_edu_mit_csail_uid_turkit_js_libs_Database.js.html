  2 /**
  3  * You probably want to use the global variable <code>database</code>.
  4  * 
  5  * @class Each TurKit script file has a JavaScript database associated with it.
  6  *        This JavaScriptDatabase instance is called <code>database</code>.
  7  * 
  8  * <p>
  9  * You may think of a JavaScript Database as a JavaScript environment that is
 10  * persisted on disk. Any query you make to the
 11  * database is evaluated in the context of the database, the new state of the database
 12  * is written to disk, and then the result is returned.
 13  */
 14 function Database() {
 15 	this.database = javaTurKit.database
 16 }
 17 
 18 /**
 19  * A reference to the Java JavaScriptDatabase object associated with this TurKit
 20  * file.
 21  */
 22 Database.prototype.database = null
 23 
 24 /**
 25  * Evaluates <i>s</i> in the context of the JavaScript database, and returns a
 26  * deep clone of the result. Note that this function evaluates the string of
 27  * JSON returned from the JavaScript database, and returns the result.
 28  */
 29 Database.prototype.query = function(s) {
 30 	return eval("" + this.database.query(s))
 31 }
 32 
 33 /**
 34  * Evaluates <i>s</i> in the context of the JavaScript database, without change.
 35  * This version does not wrap your code in a function body.
 36  * It also does not guarantee that the results will be persisted on disk before the function returns,
 37  * so you should only use this for calls that read from the database.
 38  * Note that this version still evaluates the string of
 39  * JSON returned from the JavaScript database, and returns the result.
 40  */
 41 Database.prototype.queryRaw = function(s) {
 42 	return eval("" + Packages.edu.mit.csail.uid.turkit.RhinoUtil.json(this.database.queryRaw(s)))
 43 }
