 60 /**
 61  * You probably want to just use the global variable <code>traceManager</code>.
 62  * 
 63  * @class The TraceManager manages a sort of stack frame that is memoized on
 64  *        disk (using the JavaScript Bobble associated with the current file).
 65  *        All stack frames are memoized, so the result is a sort of stack tree.
 66  * 
 67  * <p>
 68  * <code>traceManager</code> is a global instance of this class.
 69  * </p>
 70  */
 71 function TraceManager() {
 72 	this.stackFramePath = ["__stackFrames",
 73 			javaTurKit.sandbox ? "sandbox" : "for-real"]
 74 	this.stackIndexes = [0]
 75 	this.visitedStackFrames = {}
 76 	this.beforePushFrame = true
 77 }
 78 
 79 /**
 80  * Pushes a new stack frame onto the memoized stack.
 81  */
 82 TraceManager.prototype.pushFrame = function(frameName) {
 83 	this.beforePushFrame = false
 84 	if (frameName) {
 85 		this.stackFramePath.push("namedFrames")
 86 		this.stackFramePath.push(frameName)
 87 	} else {
 88 		this.stackFramePath.push("sequencialFrames")
 89 		this.stackFramePath
 90 				.push(this.stackIndexes[this.stackIndexes.length - 1]++)
 91 	}
 92 	this.stackIndexes.push(0)
 93 
 94 	var path = json(this.stackFramePath)
 95 
 96 	// make sure we haven't already been here
 97 	if (this.visitedStackFrames[path]) {
 98 		throw "visiting the same stack frame twice: " + path
 99 	}
100 	this.visitedStackFrames[path] = true
101 
102 	return bobble.query("return prune(ensure(null, " + path + ", " + json({
103 				creationTime : time()
104 			}) + "))")
105 }
106 
107 /**
108  * Sets the value of a "local variable" in the current memoized stack frame.
109  * This value is persisted in the memoized stack.
110  */
111 TraceManager.prototype.setFrameValue = function(name, value) {
112 	bobble.query("ensure(null, " + json(this.stackFramePath.concat(["values"]))
113 			+ ")[" + json(name) + "] = " + json(value))
114 }
115 
116 /**
117  * Gets the value of a "local variable" in the current memoized stack frame.
118  */
119 TraceManager.prototype.getFrameValue = function(name) {
120 	bobble.query("return ensure(null, "
121 			+ json(this.stackFramePath.concat(["values"])) + ")[" + json(name)
122 			+ "]")
123 }
124 
125 /**
126  * Pops the top stack frame off of the memoized stack. Note that this does not
127  * remove anything from the memoized stack. The memoized stack is a tree which
128  * stores a trace of every stack frame that ever went onto the stack.
129  */
130 TraceManager.prototype.popFrame = function() {
131 	this.stackFramePath.pop()
132 	this.stackFramePath.pop()
133 	this.stackIndexes.pop()
134 }
135 
136 /**
137  * Calls the function <i>func</i> only once; subsequent runs of the program
138  * will not call <i>func</i> if it returns successfully this time. If <i>func</i>
139  * throws an exception, then it will be called again when the program is
140  * re-executed.
141  * 
142  * <p>
143  * This function creates a new memoized stack frame before calling <i>func</i>.
144  * You can give this stack frame a name as an optional second parameter
145  * <i>frameName</i>.
146  * </p>
147  */
148 TraceManager.prototype.once = function(func, frameName) {
149 	var frame = this.pushFrame(frameName)
150 	try {
151 		if ("returnValue" in frame) {
152 			frame.returnValue = bobble.query("return ensure(null, "
153 					+ json(this.stackFramePath) + ").returnValue")
154 		}
155 
156 		else {
157 			frame = {
158 				returnValue : func(),
159 				returnTime : time()
160 			}
161 			bobble.query("merge(ensure(null, " + json(this.stackFramePath)
162 					+ "), " + json(frame) + ")")
163 		}
164 	} finally {
165 		this.popFrame()
166 	}
167 	return frame.returnValue
168 }
169 
170 /**
171  * This is a wrapper around {@link TraceManager#once}.
172  */
173 function once(func, frameName) {
174 	return traceManager.once(func, frameName)
175 }
176 
177 /**
178  * Calls the function <i>func</i> in a new memoized stack frame, and catches
179  * the "stop" exception. Returns <code>true</code> if the call succeeds.
180  * Returns <code>false</code> if <i>func</i> throws a "stop" exception (see
181  * {@link TraceManager#stop}).
182  */
183 TraceManager.prototype.attempt = function(func, frameName) {
184 	this.pushFrame(frameName)
185 	try {
186 		func()
187 	} catch (e) {
188 		if ("" + e == "stop") {
189 			return false
190 		} else {
191 			throw e
192 		}
193 	} finally {
194 		this.popFrame()
195 	}
196 	return true
197 }

200  * This is a wrapper around {@link TraceManager#attempt}.
201  */
202 function attempt(func, frameName) {
203 	return traceManager.attempt(func, frameName)
204 }

206 /**
207  * Sets the root of the memoized stack frame tree. Calling this method with a
208  * new value for <i>traceName</i> has the effect of reseting all calls to
209  * {@link TraceManager#once}, so that they re-execute their functions.
210  */
211 TraceManager.prototype.setTrace = function(traceName) {
212 	if (!this.beforePushFrame) {
213 		throw "you may not call setTrace from within a once or attempt, or after pushing a frame"
214 	}
215 	if (!traceName) {
216 		throw "you must provide an identifier or version number of some sort"
217 	}
218 	this.stackFramePath[1] = (javaTurKit.sandbox ? "sandbox" : "for-real")
219 			+ ":" + traceName
220 }

223  * Wrapper around {@link TraceManager#setTrace}
224  */
225 function setTrace(traceName) {
226 	return traceManager.setTrace(traceName)
227 }

229 /**
230  * Similar to {@link TraceManager#setTrace}, except that it clears the memoized
231  * stack frame data for all other trace versions.
232  */
233 TraceManager.prototype.resetTrace = function(traceName) {
234 	if (!this.beforePushFrame) {
235 		throw "you may not call resetTrace from within a once or attempt, or after pushing a frame"
236 	}
237 	if (!traceName) {
238 		throw "you must provide an identifier or version number of some sort"
239 	}
240 	this.stackFramePath[1] = (javaTurKit.sandbox ? "sandbox" : "for-real")
241 			+ ":" + traceName
242 	bobble.query("var a = " + this.stackFramePath[0]
243 			+ "; foreach(a, function (v, k) {if (k != "
244 			+ json(this.stackFramePath[1]) + ") {delete a[k]}})")
245 }

302 /*
303  * Throw an exception if we cannot spend the given amount of <i>money</i>, or
304  * create the given number of <i>hits</i>, without violating our safety limits.
305  * The <i>callbackBeforeCrash</i> will be called before an exception is thrown,
306  * if an exception is about to be thrown.
307  */
308 MTurk.prototype.assertWeCanSpend = function(money, hits, callbackBeforeCrash) {
309 	if (!javaTurKit.safety)
310 		return
311 
312 	var safety = {
313 		moneySpent : 0,
314 		hitsCreated : 0
315 	}
316 	safety = bobble.query("return ensure('__safetyCounters', " + json(safety)
317 			+ ")")
318 	safety.moneySpent += money
319 	safety.hitsCreated += hits
320 	if (safety.moneySpent > javaTurKit.maxMoney) {
321 		callbackBeforeCrash()
322 		throw "TurKit has detected a safety violation: spending too much money."
323 				+ "You need to increase your spending limit with TurKit (not with MTurk) to overcome this problem."
324 	}
325 	if (safety.hitsCreated > javaTurKit.maxHITs) {
326 		callbackBeforeCrash()
327 		throw "TurKit has detected a safety violation: creating too many HITs."
328 				+ "You need to increase your hit limit with TurKit (not with MTurk) to overcome this problem."
329 	}
330 	safety = bobble.query("__safetyCounters = " + json(safety))
331 }

333 /**
334  * Repeatedly tries to execute the given <i>func</i>, and returns the result
335  * once it succeeds.
336  */
337 MTurk.prototype.keepTrying = function(func) {
338 	sleep(0.05)
339 	var waitTime = 0.1
340 	for (var i = 0; i < 10; i++) {
341 		try {
342 			var ret = func()
343 			return ret
344 		} catch (e) {
345 			if (/throttled/.exec("" + e)) {
346 				verbosePrint("throttled")
347 				sleep(waitTime)
348 				waitTime += 0.1
349 			} else {
350 				throw e
351 			}
352 		}
353 	}
354 	// try one last time (this time don't catch the error)
355 	return func()
356 }

367 /**
368  * Creates a HIT. <i>params</i> is an object with the following values:
369  * <ul>
370  * <li><b>title</b>: displayed in the list of HITs on MTurk.</li>
371  * <li><b>description</b>: <b>desc</b> is also accepted. A slightly longer
372  * description of the HIT, also shown in the list of HITs on MTurk</li>
373  * <li><b>question</b>: a string of XML specifying what will be shown. <a
374  * href="http://docs.amazonwebservices.com/AWSMechanicalTurkRequester/2008-08-02/index.html?ApiReference_QuestionFormDataStructureArticle.html">See
375  * documentation here</a>.</li>
376  * <li><b>reward</b>: how many dollars you want to pay per assignment for this
377  * HIT.
378  * </ul>
379  * The following values are optional:
380  * <ul>
381  * <li><b>hitTypeId</b>: this may be given instead of the information above;
382  * usually it is not known or needed.</li>
383  * <li><b>keywords</b>: keywords to help people search for your HIT.</li>
384  * <li><b>assignmentDurationInSeconds</b>: default is 1 hour's worth of
385  * seconds.</li>
386  * <li><b>autoApprovalDelayInSeconds</b>: default is 1 month's worth of
387  * seconds.</li>
388  * <li><b>lifetimeInSeconds</b>: default is 1 week's worth of seconds.</li>
389  * <li><b>maxAssignments</b>: default is 1.</li>
390  * <li><b>requesterAnnotation</b>: default is no annotation.</li>
391  * <li><b>qualificationRequirements</b>: default is no requirements.</li>
392  * <li><b>minApproval</b>: minimum approval percentage. The appropriate
393  * requirement will be added if you supply a percentage here.</li>
394  * <li><b>responseGroup</b>: default is no response group.</li>
395  * </ul>
396  */
397 MTurk.prototype.createHITRaw = function(params) {
398 	if (!params)
399 		params = {}
400 
401 	if (params.HITTypeID)
402 		params.hitTypeId = params.HITTypeID
403 	if (params.hitTypeId) {
404 	} else {
405 		params.hitTypeId = null
406 		if (!params.title)
407 			throw "createHIT requires a title"
408 
409 		if (params.desc)
410 			params.description = params.desc
411 		if (!params.description)
412 			throw "createHIT requires a description"
413 
414 		if (!params.reward)
415 			throw "createHIT requires a reward"
416 
417 		if (!params.assignmentDurationInSeconds)
418 			params.assignmentDurationInSeconds = 60 * 60 // one hour
419 
420 		if (params.minApproval) {
421 			ensure(params, "qualificationRequirements", [])
422 					.push(new Packages.com.amazonaws.mturk.requester.QualificationRequirement(
423 							"000000000000000000L0",
424 							Packages.com.amazonaws.mturk.requester.Comparator.GreaterThanOrEqualTo,
425 							params.minApproval, null, false))
426 		}
427 	}
428 
429 	if (params.url) {
430 		if (!params.height)
431 			params.height = 600
432 
433 		params.question = '<ExternalQuestion xmlns="http://mechanicalturk.amazonaws.com/AWSMechanicalTurkDataSchemas/2006-07-14/ExternalQuestion.xsd">'
434 				+ '<ExternalURL>'
435 				+ escapeXml(params.url)
436 				+ '</ExternalURL>'
437 				+ '<FrameHeight>'
438 				+ params.height
439 				+ '</FrameHeight>'
440 				+ '</ExternalQuestion>'
441 	}
442 
443 	if (!params.question)
444 		throw "createHIT requires a question"
445 
446 	if (!params.lifetimeInSeconds)
447 		params.lifetimeInSeconds = 60 * 60 * 24 * 7 // one week
448 
449 	if (!params.maxAssignments)
450 		params.maxAssignments = 1
451 
452 	if (!params.autoApprovalDelayInSeconds)
453 		params.autoApprovalDelayInSeconds = null
454 
455 	if (javaTurKit.safety) {
456 		if (params.hitTypeId) {
457 			ensure(params, 'responseGroup', []).push('HITDetail')
458 		} else {
459 			assertWeCanSpend(params.reward * params.maxAssignments,
460 					params.maxAssignments)
461 		}
462 	}
463 
464 	if (!params.requesterAnnotation)
465 		params.requesterAnnotation = null
466 
467 	if (!params.responseGroup)
468 		params.responseGroup = null
469 
470 	if (!params.qualificationRequirements)
471 		params.qualificationRequirements = null
472 
473 	var hit = keepTrying(function() {
474 				return this.requesterService.createHIT(params.hitTypeId,
475 						params.title, params.description, params.keywords,
476 						params.question, params.reward,
477 						params.assignmentDurationInSeconds,
478 						params.autoApprovalDelayInSeconds,
479 						params.lifetimeInSeconds, params.maxAssignments,
480 						params.requesterAnnotation,
481 						params.qualificationRequirements, params.responseGroup);
482 			})
483 
484 	if (javaTurKit.safety) {
485 		if (params.hitTypeId) {
486 			assertWeCanSpend(parseFloat(hit.getReward().getAmount())
487 							* params.maxAssignments, params.maxAssignments,
488 					function() {
489 						disableHITRaw(hit)
490 					})
491 		} else {
492 		}
493 	}
494 
495 	var hitId = "" + hit.getHITId()
496 	verbosePrint("created HIT: " + hitId)
497 	verbosePrint("        url: "
498 			+ (javaTurKit.sandbox
499 					? "https://workersandbox.mturk.com/mturk/preview?groupId="
500 					: "https://www.mturk.com/mturk/preview?groupId=")
501 			+ hit.getHITTypeId())
502 	return hitId
503 }

734 /**
735  * Returns an object representing all of the information MTurk has on the given
736  * HIT, including the assigments, and the associated answer data. The returned
737  * object will have a value called <i>done</i> set to true iff all the pending
738  * assigments for this HIT have been completed.
739  * 
740  * <p>
741  * Note that the <i>answer</i> data structure associated with each assigment is
742  * simplified. It is recommended that you print out the result of this function
743  * using {@link json}, in order to know what it looks like for your specific
744  * situation.
745  * </p>
746  */
747 MTurk.prototype.getHIT = function(hit) {
748 	var hitId = tryToGetHITId(hit)
749 	var hit = keepTrying(function() {
750 				return this.requesterService.getHIT(hitId)
751 			})
752 	var hit = {
753 		hitId : "" + hit.getHITId(),
754 		hitTypeId : "" + hit.getHITTypeId(),
755 		title : "" + hit.getTitle(),
756 		description : "" + hit.getDescription(),
757 		keywords : "" + hit.getKeywords(),
758 		reward : "" + hit.getReward().getAmount(),
759 		question : hit.getQuestion(),
760 		maxAssignments : "" + hit.getMaxAssignments(),
761 		assignmentDurationInSeconds : hit.getAssignmentDurationInSeconds(),
762 		autoApprovalDelayInSeconds : hit.getAutoApprovalDelayInSeconds(),
763 		requesterAnnotation : hit.getRequesterAnnotation(),
764 		hitStatus : "" + hit.getHITStatus(),
765 		hitReviewStatus : "" + hit.getHITReviewStatus(),
766 		creationTime : hit.getCreationTime().getTimeInMillis(),
767 		expiration : hit.getExpiration().getTimeInMillis(),
768 		assignments : []
769 	}
770 	foreach(getAssignmentsForHIT(hitId), function(javaAssignment) {
771 				function getTime(t) {
772 					if (t != null)
773 						return t.getTimeInMillis()
774 				}
775 				var assignment = {
776 					assignmentId : "" + javaAssignment.getAssignmentId(),
777 					hitId : "" + javaAssignment.getHITId(),
778 					assignmentStatus : ""
779 							+ javaAssignment.getAssignmentStatus(),
780 					requesterFeedback : ""
781 							+ javaAssignment.getRequesterFeedback(),
782 					workerId : "" + javaAssignment.getWorkerId(),
783 					acceptTime : getTime(javaAssignment.getAcceptTime()),
784 					approvalTime : getTime(javaAssignment.getApprovalTime()),
785 					autoApprovalTime : getTime(javaAssignment
786 							.getAutoApprovalTime()),
787 					deadline : getTime(javaAssignment.getDeadline()),
788 					rejectionTime : getTime(javaAssignment.getRejectionTime()),
789 					submitTime : getTime(javaAssignment.getSubmitTime())
790 				}
791 
792 				// deal with answers
793 				// NOTE: we use the name "answer" to refer to the set of
794 				// answers, because that is how Mechanical Turk refers to them
795 				// in the
796 				// Assignment data structure
797 				assignment.answer = {}
798 				var answers = this.requesterService.parseAnswers(javaAssignment
799 						.getAnswer()).getAnswer()
800 				for (var ai = 0; ai < answers.size(); ai++) {
801 					var a = answers.get(ai)
802 
803 					var rhs = a.getFreeText()
804 					if (rhs) {
805 						rhs = "" + rhs
806 					}
807 					if (!rhs) {
808 						rhs = a.getUploadedFileKey()
809 						if (rhs) {
810 							rhs = {
811 								uploadedFileKey : "" + rhs,
812 								uploadedFileSizeInBytes : a
813 										.getUploadedFileSizeInBytes()
814 							}
815 						}
816 					}
817 					if (!rhs) {
818 						rhs = []
819 						var sels = a.getSelectionIdentifier()
820 						for (var si = 0; si < sels.size(); si++) {
821 							rhs.push("" + sels.get(si))
822 						}
823 						var o = a.getOtherSelectionText()
824 						if (o) {
825 							rhs.push("" + o)
826 						}
827 					}
828 
829 					assignment.answer["" + a.getQuestionIdentifier()] = rhs
830 				}
831 
832 				hit.assignments.push(assignment)
833 			})
834 	hit.done = (hit.hitStatus == "Reviewable")
835 			&& (hit.assignments.length == hit.maxAssignments)
836 	return hit
837 }

839 /**
840  * Returns information about the <code>hit</code> if it is done (see
841  * {@link MTurk#getHIT}), and throwing the "stop" exception if it is still in
842  * progress.
843  */
844 MTurk.prototype.waitForHIT = function(hit) {
845 	var hitId = tryToGetHITId(hit)
846 	return once(function() {
847 				var hit = getHIT(hitId);
848 				if (!hit.done) {
849 					stop()
850 				};
851 				verbosePrint("hit completed: " + hitId)
852 				return hit
853 			})
854 }

859 /**
860  * Collects multiple votes until a minimum number of necessary votes for a
861  * single choice is achieved. The <code>hit</code> supplied to this function
862  * must have its <i>maxAssignments</i> set to the minimum number of votes
863  * necessary for a single choice. This function will add assignments if
864  * necessary. This function relies on <code>extractVoteFromAnswer</code> to
865  * extract a String representing the choice given an answer data structure (see
866  * {@link MTurk#getHIT}).
867  * 
868  * <p>
869  * The return value is an object with the following entries:
870  * </p>
871  * <ul>
872  * <li><b>bestOption</b>: the String representing the choice with the most
873  * votes.</li>
874  * <li><b>totalVoteCount</b>: the number of votes received from turkers.</li>
875  * <li><b>voteCounts</b>: an object with key/value pairs representing
876  * choice/voteCounts.</li>
877  * </ul>
878  */
879 MTurk.prototype.vote = function(hit, extractVoteFromAnswer) {
880 	var necessaryVoteCount = null
881 	var hitId = tryToGetHITId(hit)
882 	while (true) {
883 		var hit = waitForHIT(hitId)
884 
885 		if (necessaryVoteCount == null) {
886 			necessaryVoteCount = hit.maxAssignments
887 		}
888 
889 		var votes = {}
890 		foreach(hit.assignments, function(assignment) {
891 					var vote = extractVoteFromAnswer(assignment.answer)
892 					votes[vote] = ensure(votes, [vote], 0) + 1
893 				})
894 		var winnerVotes, winner
895 		[winnerVotes, winner] = getMax(votes)
896 
897 		if (winnerVotes >= necessaryVoteCount) {
898 			foreach(hit.assignments, function(assignment) {
899 						approveAssignment(assignment)
900 					})
901 			deleteHIT(hit)
902 			return {
903 				bestOption : winner,
904 				totalVoteCount : hit.assignments.length,
905 				voteCounts : votes
906 			}
907 		} else {
908 			extendHIT(hit, necessaryVoteCount - winnerVotes, null)
909 		}
910 	}
911 }

913 /**
914  * Works just like the JavaScript array sort function, except that this one can
915  * perform comparisons in parallel on MTurk by catching the "stop" exception
916  * which is thrown when waiting on an MTurk HIT using {@link MTurk#waitForHIT}.
917  */
918 MTurk.prototype.sort = function(a, comparator) {
919 	traceManager.pushFrame()
920 	try {
921 		var sortTree = getFrameValue("sortTree")
922 		if (!sortTree)
923 			sortTree = {}
924 
925 		function insertIntoTree(index, tree) {
926 			if (tree.index == null) {
927 				tree.index = index
928 				return
929 			}
930 			var comp = comparator(a[index], a[tree.index])
931 			if (comp == 0) {
932 			} else if (comp < 0) {
933 				insertIntoTree(index, ensure(tree, ["left"], {}))
934 			} else {
935 				insertIntoTree(index, ensure(tree, ["right"], {}))
936 			}
937 		}
938 		var done = true
939 		foreach(a, function(e, i) {
940 					if (attempt(function() {
941 								insertIntoTree(i, sortTree)
942 							})) {
943 						setFrameValue("sortTree", sortTree)
944 					} else {
945 						done = false
946 					}
947 				})
948 		if (!done)
949 			stop()
950 
951 		var newA = []
952 		function traverseTree(tree) {
953 			if (!tree)
954 				return
955 			traverseTree(tree.left)
956 			newA.push(a[tree.index])
957 			traverseTree(tree.right)
958 		}
959 		traverseTree(sortTree)
960 		return newA
961 	} finally {
962 		traceManager.popFrame()
963 	}
964 }

985 /**
986  * Randomizes the order of the elements in <i>a</i>.
987  */
988 function shuffle(a) {
989 	for (var i = 0; i < a.length; i++) {
990 		var ii = Math.floor(random() * a.length)
991 		var temp = a[i]
992 		a[i] = a[ii]
993 		a[ii] = temp
994 	}
995 	return a
996 }
