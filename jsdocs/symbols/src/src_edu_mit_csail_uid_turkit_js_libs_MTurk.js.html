 15 /*
 16  * Throw an exception if we cannot spend the given amount of <i>money</i>, or
 17  * create the given number of <i>hits</i>, without violating our safety limits.
 18  * The <i>callbackBeforeCrash</i> will be called before an exception is thrown,
 19  * if an exception is about to be thrown.
 20  */
 21 MTurk.prototype.assertWeCanSpend = function(money, hits, callbackBeforeCrash) {
 22 	if (!javaTurKit.safety)
 23 		return
 24 
 25 	var safety = {
 26 		moneySpent : 0,
 27 		hitsCreated : 0
 28 	}
 29 	safety = database.query("return ensure('__safetyCounters', " + json(safety)
 30 			+ ")")
 31 	safety.moneySpent += money
 32 	safety.hitsCreated += hits
 33 	if (safety.moneySpent > javaTurKit.maxMoney) {
 34 		if (callbackBeforeCrash)
 35 			callbackBeforeCrash()
 36 		throw new java.lang.Exception("TurKit has detected a safety violation: spending too much money. "
 37 				+ "You need to increase your spending limit with TurKit (not with MTurk) to overcome this problem.")
 38 	}
 39 	if (safety.hitsCreated > javaTurKit.maxHITs) {
 40 		if (callbackBeforeCrash)
 41 			callbackBeforeCrash()
 42 		throw new java.lang.Exception("TurKit has detected a safety violation: creating too many HITs. "
 43 				+ "You need to increase your hit limit with TurKit (not with MTurk) to overcome this problem.")
 44 	}
 45 	safety = database.query("__safetyCounters = " + json(safety))
 46 }

 57 /**
 58  * Creates a HIT. <i>params</i> is an object with the following properties:
 59  * <ul>
 60  * <li><b>title</b>: displayed in the list of HITs on MTurk.</li>
 61  * <li><b>description</b>: <b>desc</b> is also accepted. A slightly longer
 62  * description of the HIT, also shown in the list of HITs on MTurk</li>
 63  * <li><b>question</b>: a string of XML specifying what will be shown. <a
 64  * href="http://docs.amazonwebservices.com/AWSMechanicalTurkRequester/2008-08-02/index.html?ApiReference_QuestionFormDataStructureArticle.html">See
 65  * documentation here</a>. Instead of <i>question</i>, you may use the following special parameters:
 66  <ul>
 67  <li><b>url</b>: creates an external question pointing to this URL</li>
 68  <li><b>height</b>: (optional) height of the iFrame embedded in MTurk, in pixels (default is 600).</li>
 69  </ul>
 70  
 71  </li>
 72  * <li><b>reward</b>: how many dollars you want to pay per assignment for this
 73  * HIT.
 74  * </ul>
 75  * The following properties are optional:
 76  * <ul>
 77  * <li><b>keywords</b>: keywords to help people search for your HIT.</li>
 78  * <li><b>assignmentDurationInSeconds</b>: default is 1 hour's worth of
 79  * seconds.</li>
 80  * <li><b>autoApprovalDelayInSeconds</b>: default is 1 month's worth of
 81  * seconds.</li>
 82  * <li><b>lifetimeInSeconds</b>: default is 1 week's worth of seconds.</li>
 83  * <li><b>maxAssignments</b>: <b>assignments</b> is also accepted. default is 1.</li>
 84  * <li><b>requesterAnnotation</b>: default is no annotation.</li>
 85  * <li><b>qualificationRequirements</b>: default is no requirements.</li>
 86  * <li><b>minApproval</b>: minimum approval percentage. The appropriate
 87  * requirement will be added if you supply a percentage here.</li>
 88  * </ul>
 89  */
 90 MTurk.prototype.createHITRaw = function(params) {
 91 	if (!params)
 92 		params = {}
 93 		
 94 	// let them know if they provide param names that are not on the list
 95 	var badKeys = keys(new Set(keys(params)).remove([
 96 		"title",
 97 		"desc",
 98 		"description",
 99 		"reward",
100 		"assignmentDurationInSeconds",
101 		"minApproval",
102 		"html",
103 		"bucket",
104 		"url",
105 		"blockWorkers",
106 		"height",
107 		"question",
108 		"lifetimeInSeconds",
109 		"assignments",
110 		"maxAssignments",
111 		"numAssignments",
112 		"autoApprovalDelayInSeconds",
113 		"requesterAnnotation",
114 		"keywords",
115 		"qualificationRequirements",
116 	]))
117 	if (badKeys.length > 0) {
118 		throw new java.lang.Exception("some parameters to createHIT are not understood: " + badKeys.join(', '))
119 	}
120 
121 	if (!params.title)
122 		throw new java.lang.Exception("createHIT requires a title")
123 
124 	if (params.desc)
125 		params.description = params.desc
126 	if (!params.description)
127 		throw new java.lang.Exception("createHIT requires a description")
128 
129 	if (params.reward == null)
130 		throw new java.lang.Exception("createHIT requires a reward")
131 
132 	if (!params.assignmentDurationInSeconds)
133 		params.assignmentDurationInSeconds = 60 * 60 // one hour
134 
135 	if (params.minApproval) {
136 		var q = ensure(params, "qualificationRequirements", [])
137 		var i = (q.length / 3) + 1
138 		q.push("QualificationRequirement." + i + ".QualificationTypeId")
139 		q.push("000000000000000000L0")
140 		q.push("QualificationRequirement." + i + ".Comparator")
141 		q.push("GreaterThanOrEqualTo")
142 		q.push("QualificationRequirement." + i + ".IntegerValue")
143 		q.push(params.minApproval)
144 	}
145 
146 	if (params.html) {
147 		if (!params.bucket) {
148 			params.bucket = javaTurKit.awsAccessKeyID.toLowerCase() + "-turkit"
149 		}
150 		var s = ("" + javaTurKit.taskTemplate).replace(/\[\[\[CONTENT\]\]\]/, params.html)
151 		var key = Packages.edu.mit.csail.uid.turkit.util.U.md5(s) + ".html"
152 		params.url = s3.putString(params.bucket, key, s)
153 		
154 		if (params.blockWorkers) {
155 			if ((typeof params.blockWorkers) != "string") {
156 				params.blockWorkers = params.blockWorkers.join(",")
157 			}
158 			params.url += "?blockWorkers=" + params.blockWorkers
159 		}
160 	}
161 	
162 	if (params.url) {
163 		if (!params.height)
164 			params.height = 600
165 
166 		params.question = '<ExternalQuestion xmlns="http://mechanicalturk.amazonaws.com/AWSMechanicalTurkDataSchemas/2006-07-14/ExternalQuestion.xsd">'
167 				+ '<ExternalURL>'
168 				+ escapeXml(params.url)
169 				+ '</ExternalURL>'
170 				+ '<FrameHeight>'
171 				+ params.height
172 				+ '</FrameHeight>'
173 				+ '</ExternalQuestion>'
174 	}
175 
176 	if (!params.question)
177 		throw new java.lang.Exception("createHIT requires a question (or a url, or html)")
178 
179 	if (!params.lifetimeInSeconds)
180 		params.lifetimeInSeconds = 60 * 60 * 24 * 7 // one week
181 		
182 	if (params.assignments)
183 		params.maxAssignments = params.assignments
184 	if (params.numAssignments)
185 		params.maxAssignments = params.numAssignments 
186 	if (!params.maxAssignments)
187 		params.maxAssignments = 1
188 
189 	if (!params.autoApprovalDelayInSeconds)
190 		params.autoApprovalDelayInSeconds = null
191 
192 	if (javaTurKit.safety) {
193 		if (params.hitTypeId) {
194 			ensure(params, 'responseGroup', []).push('HITDetail')
195 		} else {
196 			this.assertWeCanSpend(params.reward * params.maxAssignments, 1)
197 		}
198 	}
199 
200 	if (!params.requesterAnnotation)
201 		params.requesterAnnotation = null
202 
203 	if (!params.responseGroup)
204 		params.responseGroup = null
205 
206 	if (!params.qualificationRequirements)
207 		params.qualificationRequirements = null
208 
209 	var x = new XML(javaTurKit.restRequest("CreateHIT",
210 			["Title", params.title,
211 			"Description", params.description,
212 			"Question", params.question,
213 			"Reward.1.Amount", params.reward,
214 			"Reward.1.CurrencyCode", "USD",
215 			"AssignmentDurationInSeconds", params.assignmentDurationInSeconds,
216 			"LifetimeInSeconds", params.lifetimeInSeconds,
217 			"Keywords", params.keywords,
218 			"MaxAssignments", params.maxAssignments,
219 			"AutoApprovalDelayInSeconds", params.autoApprovalDelayInSeconds,
220 			"RequesterAnnotation", params.requesterAnnotation].
221 			concat(params.qualificationRequirements ? params.qualificationRequirements : [])
222 		))
223 	if ('' + x..Request.IsValid != "True") throw "Failed to create HIT: " + x
224 	var hit = x..HIT
225 
226 	var hitId = this.tryToGetHITId(hit)
227 	
228 	verbosePrint("created HIT: " + hitId)
229 	var url = (javaTurKit.mode == "sandbox"
230 					? "https://workersandbox.mturk.com/mturk/preview?groupId="
231 					: "https://www.mturk.com/mturk/preview?groupId=")
232 			+ hit.HITTypeId
233 	verbosePrint("        url: " + url)
234 	database.query("ensure(null, ['__HITs', " + json(javaTurKit.mode + ":" + hitId) + "], " + json({url : url}) + ")")
235 	return hitId
236 }

247 /**
248  * Returns a list of HIT Ids of HITs that are ready to be reviewed.
249  * You may optionally specify <code>maxPages</code>,
250  * to limit the number of pages of results returned.
251  * Each page will have up to 100 reviewable HIT Ids.
252  * If <code>maxPages</code> is specified,
253  * then the return value will have a property called <code>totalNumResults</code>,
254  * which indicates how many HITs are reviewable.
255  */
256 MTurk.prototype.getReviewableHITs = function(maxPages) {
257     var all = []
258     var page = 1
259     var processedResults = 0
260     var totalNumResults = 0
261     while (!maxPages || (page <= maxPages)) {
262         var x = new XML(javaTurKit.restRequest("GetReviewableHITs",
263             "SortProperty", "CreationTime",
264             "PageSize", "100",
265             "PageNumber", "" + page))
266         if ('' + x..Request.IsValid != "True") throw "GetReviewableHITs failed: " + x
267         foreach(x..HITId, function (hitId) {
268         	all.push("" + hitId)
269         })
270         var numResults = parseInt(x..NumResults)
271         if (numResults <= 0) break
272         processedResults += numResults
273         totalNumResults = parseInt(x..TotalNumResults)
274         if (processedResults >= totalNumResults) break
275         page++
276     }
277     if (maxPages) {
278     	all.totalNumResults = totalNumResults
279     }
280     return all
281 }

283 /**
284  * Tries to determine the type of <code>hit</code>, and return the HIT Id.
285  */
286 MTurk.prototype.tryToGetHITId = function(hit) {
287 	if ((typeof hit) == "xml") {
288 		return '' + hit..HITId
289 	} else if ((typeof hit) == "object") {
290 		try {
291 			if (hit.hitId) {
292 				return hit.hitId
293 			}
294 		} catch (e) {
295 		}
296 		return "" + hit
297 	}
298 	return hit
299 }

320 /*
321 input: a string representing a date/time from MTurk
322 example: 2010-01-04T01:17:16Z
323 output: UNIX time for this time
324  */
325 function parseMTurkTime(s) {
326     var m = s.match(/(\d+)-(\d+)-(\d+)T([0-9:]+)Z/)
327     return Date.parse(m[2] + "/" + m[3] + "/" + m[1] + " " + m[4] + " GMT")
328 }

330 /*
331 input: a HIT XML element from MTurk
332 output: a JS object representing the HIT
333  */
334 MTurk.prototype.parseHIT = function(hit) {
335 
336     var xmlNull = new XML()..blah
337     function g(x) {
338         if (x == xmlNull) return null
339         return "" + x
340     }
341     function gi(x) {
342         if (x == xmlNull) return null
343         return parseInt(x)
344     }
345     function gf(x) {
346         if (x == xmlNull) return null
347         return parseFloat(x)
348     }
349     function gt(x) {
350         if (x == xmlNull) return null
351         return parseMTurkTime(x)
352     }
353     
354 	return {
355 		hitId : g(hit.HITId),
356 		hitTypeId : g(hit.HITTypeId),
357 		title : g(hit.Title),
358 		description : g(hit.Description),
359 		keywords : g(hit.Keywords),
360 		reward : gf(hit.Reward.Amount),
361 		question : g(hit.Question),
362 		maxAssignments : parseInt(hit.MaxAssignments),
363 		assignmentDurationInSeconds : gi(hit.AssignmentDurationInSeconds),
364         autoApprovalDelayInSeconds : gi(hit.AutoApprovalDelayInSeconds),
365 		requesterAnnotation : g(hit.RequesterAnnotation),
366 		hitStatus : g(hit.HITStatus),
367 		hitReviewStatus : g(hit.HITReviewStatus),
368 		creationTime : gt(hit.CreationTime),
369 		expiration : gt(hit.Expiration)
370 	}
371 }

373 /**
374  * Returns an object representing all of the information MTurk has on the given
375  * HIT, including the assigments, and the associated answer data. The returned
376  * object will have a value called <i>done</i> set to true iff all the pending
377  * assigments for this HIT have been completed (unless <code>getAssignments</code> is false).
378  * 
379  * <p>
380  * Note that the <i>answer</i> data structure associated with each assigment is
381  * simplified. It is recommended that you print out the result of this function
382  * using {@link json}, in order to know what it looks like for your specific
383  * situation.
384  * </p>
385  */
386 MTurk.prototype.getHIT = function(hit, getAssignments) {
387 
388 	if (getAssignments === undefined) getAssignments = true
389 
390 	var hitId = this.tryToGetHITId(hit)
391 	var x = new XML(javaTurKit.restRequest("GetHIT", "HITId", hitId))
392     if (x..Request.IsValid.toString() != "True") throw "GetHIT failed"
393     hit = x..HIT
394     
395     var xmlNull = new XML()..blah
396     function g(x) {
397         if (x == xmlNull) return null
398         return "" + x
399     }
400     function gi(x) {
401         if (x == xmlNull) return null
402         return parseInt(x)
403     }
404     function gf(x) {
405         if (x == xmlNull) return null
406         return parseFloat(x)
407     }
408     function gt(x) {
409         if (x == xmlNull) return null
410         return parseMTurkTime(x)
411     }
412     
413     hit = this.parseHIT(hit)
414     
415     if (!getAssignments) return hit
416     
417     hit.assignments = []
418     
419     function processAssignment(a) {
420         // NOTE: we use the singular "answer" instead of "answers"
421         // to be consistent with MTurk
422         var answer = {}
423     
424         var answers = a.Answer
425         answers = answers.substring(answers.indexOf("?>\n") + 3)
426         answers = new XML(answers)
427         foreach(answers.*::Answer, function (a) {
428             var rhs = g(a.*::FreeText)
429             if (rhs == null) {
430                 rhs = g(a.*::UploadedFileKey)
431                 if (rhs != null) {
432                     rhs = {
433                         uploadedFileKey : rhs,
434                         uploadedFileSizeInBytes : gi(a.*::UploadedFileSizeInBytes)
435                     }
436                 }
437             }
438             if (rhs == null) {
439                 rhs = []
440                 foreach(a.*::SelectionIdentifier, function (sel) {
441                     rhs.push("" + sel)
442                 })
443                 var o = g(a.*::OtherSelectionText)
444                 if (o != null) {
445                     rhs.push(o)
446                 }
447             }
448 
449             answer[g(a.*::QuestionIdentifier)] = rhs
450         })
451     
452         hit.assignments.push({
453             assignmentId : g(a.AssignmentId),
454             workerId : g(a.WorkerId),
455             hitId : g(a.HITId),
456             assignmentStatus : g(a.AssignmentStatus),
457             autoApprovalTime : gt(a.AutoApprovalTime),
458             acceptTime : gt(a.AcceptTime),
459             submitTime : gt(a.SubmitTime),
460             answer : answer,
461             requesterFeedback : g(a.RequesterFeedback),
462             approvalTime : gt(a.ApprovalTime),
463             deadline : gt(a.Deadline),
464             rejectionTime : gt(a.RejectionTime)
465         })
466     }
467     
468     var page = 1
469     var processedResults = 0
470     var totalNumResults = 0
471     while (true) {
472         var x = new XML(javaTurKit.restRequest("GetAssignmentsForHIT",
473             "HITId", hitId,
474             "PageSize", "100",
475             "PageNumber", "" + page))
476         for each (a in x..Assignment) {
477             processAssignment(a)
478         }
479         var numResults = parseInt(x..NumResults)
480         if (numResults <= 0) break
481         processedResults += numResults
482         totalNumResults = parseInt(x..TotalNumResults)
483         if (processedResults >= totalNumResults) break
484         page++
485     }
486 
487 	hit.done = (hit.hitStatus == "Reviewable")
488 			&& (hit.assignments.length == hit.maxAssignments)
489 	return hit
490 }

492 /**
493  * Extends the <code>hit</code> by the given number of assignments (<code>moreAssignments</code>),
494  * and the given number of seconds (<code>moreSeconds</code>).
495  */
496 MTurk.prototype.extendHITRaw = function(hit, moreAssignments, moreSeconds) {
497 	if (javaTurKit.safety) {
498 		if (moreAssignments != null) {
499 			if ((typeof hit == "object") && ("reward" in hit)) {
500 			} else {
501 				hit = this.getHIT(hit)
502 			}
503 			this.assertWeCanSpend(parseFloat(hit.reward) * moreAssignments, 0)
504 		}
505 	}
506 
507 	var hitId = this.tryToGetHITId(hit)
508 	var params = ["HITId", hitId]
509     if (moreAssignments) {
510         params.push("MaxAssignmentsIncrement")
511         params.push("" + moreAssignments)
512     }
513     if (moreSeconds) {
514         params.push("ExpirationIncrementInSeconds")
515         params.push(moreSeconds)
516     }
517 	var x = new XML(javaTurKit.restRequest("ExtendHIT", params))
518     if (x..Request.IsValid.toString() != "True") throw "GetHIT failed"
519 	verbosePrint("extended HIT: " + hitId)
520 }

531 /**
532 	Deletes the given <code>hit</code>.
533 	If there are any completed assignments that have not been approved or rejected, then they are approved.
534  */
535 MTurk.prototype.deleteHITRaw = function(hit) {
536 	var hitId = this.tryToGetHITId(hit)
537     ;(function () {
538     
539         // try disabling the HIT
540         var x = new XML(javaTurKit.restRequest("DisableHIT", "HITId", hitId))
541         if (x..Request.IsValid.toString() == "True") {
542             verbosePrint("disabled HIT: " + hitId)
543             return
544         }
545         
546         // see if we already deleted the HIT
547         var hit = mturk.getHIT(hitId)
548         if (hit.hitStatus == "Disposed") {
549             verbosePrint("already deleted HIT: " + hitId)
550             return
551         }
552         
553         // ok, it must be "Reviewable"
554         // (since we couldn't disable it, and it isn't already deleted)
555         
556         // first, approve all the assignments
557         foreach(hit.assignments, function (a) {
558             if (a.assignmentStatus == "Submitted")
559                 mturk.approveAssignmentRaw(a)
560         })
561         
562         // next, dispose of the HIT
563         var x = new XML(javaTurKit.restRequest("DisposeHIT", "HITId", hitId))
564         if (x..Request.IsValid.toString() == "True") {
565             verbosePrint("disposed HIT: " + hitId)
566             return
567         }
568         
569         throw "DeleteHIT failed"
570     })()
571 	database.query("delete __HITs[" + json(javaTurKit.mode + ":" + hitId) + "]")
572 }

629 /**
630  * Approves the given <code>assignment</code>, and provides an optional
631  * <code>reason</code>.
632  */
633 MTurk.prototype.approveAssignmentRaw = function(assignment, reason) {
634 	var assignmentId = this.tryToGetAssignmentId(assignment)
635 
636 	var params = ["AssignmentId", assignmentId]
637 	if (reason) {

642     if (x..Request.IsValid.toString() != "True") throw "ApproveAssignment failed: " + x

701 /**
702  * Returns an array of HIT data for all the HITs you currently have on MTurk.
703  */
704 MTurk.prototype.getHITs = function(maxPages) {
705     var self = this
706     var all = []
707     var page = 1
708     var processedResults = 0
709     var totalNumResults = 0
710     while (!maxPages || (page <= maxPages)) {
711         var x = new XML(javaTurKit.restRequest("SearchHITs",
712             "SortProperty", "CreationTime",
713             "SortDirection", "Descending",
714             "PageSize", "100",
715             "PageNumber", "" + page))
716         if (x..Request.IsValid.toString() != "True") throw "SearchHITs failed: " + x
717         foreach(x..HIT, function (hit) {
718             all.push(self.parseHIT(hit))
719         })
720         var numResults = parseInt(x..NumResults)
721         if (numResults <= 0) break
722         processedResults += numResults
723         totalNumResults = parseInt(x..TotalNumResults)
724         if (processedResults >= totalNumResults) break
725         page++
726     }
727     if (maxPages) {
728     	a.totalNumResults = totalNumResults
729     }
730     return all
731 }

733 /**
734  * Returns information about the <code>hit</code> if it is done (see
735  * {@link MTurk#getHIT}), and throwing the "stop" exception if it is still in
736  * progress.
737  */
738 MTurk.prototype.waitForHIT = function(hit) {
739 	var me = this
740 	var hitId = this.tryToGetHITId(hit)
741 	return once(function() {
742 		// the idea of this logic
743 		// is to minimize the number of calls to MTurk
744 		// to see if HITs are done.
745 		// 
746 		// if we are going to be calling waitForHIT a lot,
747 		// then we'd like to get a list of all reviewable HITs,
748 		// and check for the current HIT against that list,
749 		// and refresh that list only if enough time has passed.
750 		//
751 		// of course, if the list of reviewable HITs is very long,
752 		// then we'd rather not retrieve it,
753 		// unless we will be calling this function a lot,
754 		// so to figure out how many times we should wait before
755 		// retrieving the list,
756 		// we start by seeing how many pages of results that list has,
757 		// and if we call this function that many times,
758 		// then we go ahead and get the list
759 
760 		if (!me.waitForHIT_callCount) {
761 			me.waitForHIT_callCount = 0
762 			var a = me.getReviewableHITs(1)
763 			if (a.totalNumResults == a.length) {
764 				me.waitForHIT_reviewableHITs = new Set(a)
765 				me.waitForHIT_reviewableHITsTime = time()
766 			}
767 			me.waitForHIT_waitCount = Math.ceil(a.totalNumResults / 100)
768 		}
769 		me.waitForHIT_callCount++
770 		if (me.waitForHIT_callCount >= me.waitForHIT_waitCount) {
771 			if (!me.waitForHIT_reviewableHITs ||
772 				(time() > me.waitForHIT_reviewableHITsTime + (1000 * 60))) {
773 				me.waitForHIT_reviewableHITs = new Set(me.getReviewableHITs())
774 				me.waitForHIT_reviewableHITsTime = time()
775 			}
776 		}
777 		if (me.waitForHIT_reviewableHITs) {
778 			if (!me.waitForHIT_reviewableHITs[hitId]) {
779 				stop()
780 			}
781 		}
782 
783 		var hit = mturk.getHIT(hitId);
784 		if (!hit.done) {
785 			stop()
786 		}
787 		verbosePrint("hit completed: " + hitId)
788 		return hit
789 	})
790 }

795 /**
796  * Collects multiple votes until a minimum number of necessary votes for a
797  * single choice is achieved. The <code>hit</code> supplied to this function
798  * must have its <i>maxAssignments</i> set to the minimum number of votes
799  * necessary for a single choice. This function will add assignments if
800  * necessary. This function relies on <code>extractVoteFromAnswer</code> to
801  * extract a String representing the choice given an answer data structure (see
802  * {@link MTurk#getHIT}).
803  * 
804  * <p>
805  * The return value is an object with the following entries:
806  * </p>
807  * <ul>
808  * <li><b>bestOption</b>: the String representing the choice with the most
809  * votes.</li>
810  * <li><b>totalVoteCount</b>: the number of votes received from turkers.</li>
811  * <li><b>voteCounts</b>: an object with key/value pairs representing
812  * choice/voteCounts.</li>
813  * <li><b>hit</b>: the HIT data structure representing the HIT on the final iteration of the voting process.</li>
814  * </ul>
815  */
816 MTurk.prototype.vote = function(hit, extractVoteFromAnswer) {
817 	var necessaryVoteCount = null
818 	var hitId = this.tryToGetHITId(hit)
819 	while (true) {
820 		var hit = this.waitForHIT(hitId)
821 
822 		if (necessaryVoteCount == null) {
823 			necessaryVoteCount = hit.maxAssignments
824 		}
825 
826 		var votes = {}
827 		foreach(hit.assignments, function(assignment) {
828 					var vote = extractVoteFromAnswer(assignment.answer)
829 					votes[vote] = ensure(votes, [vote], 0) + 1
830 				})
831 		var winnerVotes, winner
832 		[winnerVotes, winner] = getMax(votes)
833 
834 		if (winnerVotes >= necessaryVoteCount) {
835 			foreach(hit.assignments, function(assignment) {
836 						mturk.approveAssignment(assignment)
837 					})
838 			this.deleteHIT(hit)
839 			return {
840 				bestOption : winner,
841 				totalVoteCount : hit.assignments.length,
842 				voteCounts : votes,
843 				hit : hit
844 			}
845 		} else {
846 			this.extendHIT(hit, necessaryVoteCount - winnerVotes, null)
847 		}
848 	}
849 }

851 /**
852  * Works just like the JavaScript array sort function, except that this one can
853  * perform comparisons in parallel on MTurk by catching the "stop" exception
854  * which is thrown when waiting on an MTurk HIT using {@link MTurk#waitForHIT}.
855  */
856 MTurk.prototype.sort = function(a, comparator) {
857 	traceManager.pushFrame()
858 	try {
859 		var sortTree = traceManager.getFrameValue("sortTree")
860 		if (!sortTree)
861 			sortTree = {}
862 
863 		function insertIntoTree(index, tree) {
864 			if (tree.index == null) {
865 				tree.index = index
866 				return
867 			}
868 			var comp = comparator(a[index], a[tree.index])
869 			if (comp == 0) {
870 			} else if (comp < 0) {
871 				insertIntoTree(index, ensure(tree, ["left"], {}))
872 			} else {
873 				insertIntoTree(index, ensure(tree, ["right"], {}))
874 			}
875 		}
876 		var done = true
877 		foreach(a, function(e, i) {
878 					if (attempt(function() {
879 								insertIntoTree(i, sortTree)
880 							})) {
881 						traceManager.setFrameValue("sortTree", sortTree)
882 					} else {
883 						done = false
884 					}
885 				})
886 		if (!done)
887 			stop()
888 
889 		var newA = []
890 		function traverseTree(tree) {
891 			if (!tree)
892 				return
893 			traverseTree(tree.left)
894 			newA.push(a[tree.index])
895 			traverseTree(tree.right)
896 		}
897 		traverseTree(sortTree)
898 		return newA
899 	} finally {
900 		traceManager.popFrame()
901 	}
902 }
