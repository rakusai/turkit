  2 /**
  3  * Use this instead of <code>Math.random()</code>. This is not just a
  4  * shorthand. It is important that TurKit programs run the same way each time
  5  * they are executed. This function wraps <code>Math.random()</code> inside a
  6  * call to {@link TraceManager#once}.
  7  */
  8 function random() {
  9 	return once(function() {
 10 				return Math.random()
 11 			})
 12 }

 14 /**
 15  * Randomizes the order of the elements in <i>a</i>.
 16  */
 17 function shuffle(a) {
 18 	for (var i = 0; i < a.length; i++) {
 19 		var ii = Math.floor(random() * a.length)
 20 		var temp = a[i]
 21 		a[i] = a[ii]
 22 		a[ii] = temp
 23 	}
 24 	return a
 25 }

 27 /**
 28 	<p>Creates a webpage, puts it on S3, and returns the URL.
 29 	The webpage represents a HIT.
 30 	The webpage is created by starting with a <a href="/src/resources/task-template.html">template HTML page</a>.
 31 	The <code>html</code> supplied to this function in injected into the template.
 32 	The S3 webpage is made publicly accessible.
 33 	The URL to the S3 webpage is returned.
 34 	You can pass the URL to <code>s3.deleteObject</code> to remove it.</p>
 35 	
 36 	<p>If you supply an array or comma-delimited-string for <code>blockWorkers</code>,
 37 	then the html page will use JavaScript to prevent those workers from completing the HIT.</p>
 38 	<p>If you give any elements the class "random", then those elements will be randomly permuted each time a user views the page.
 39 	This is good for randomizing the choices in voting tasks.</p>
 40 	
 41 	<p><code>bucketName</code> is optional.
 42 	If you do not provide it, the name {@link S3#defaultBucketName} will be used.</p>
 43 	*/
 44 function createWebpageFromTemplate(html, blockWorkers, bucketName) {
 45 	if (!bucketName) {
 46 		bucketName = s3.defaultBucketName
 47 	}
 48 	
 49 	if (!blockWorkers) {
 50 		blockWorkers = ""
 51 	} else if (blockWorkers instanceof Array) {
 52 		blockWorkers = blockWorkers.join(",")
 53 	}
 54 	
 55 	var s = ("" + javaTurKit.taskTemplate).
 56 		replace(/___CONTENT___/, html).
 57 		replace(/___BLOCK_WORKERS___/, blockWorkers)
 58 	createWebpageFromTemplate_html = s
 59 	return s3.putString(bucketName, null, s)
 60 }

 62 /**
 63  * Changes the mode. Possible values include "offline", "sandbox" and "real".
 64  * <ul>
 65  * <li>"offline" will not let you access MTurk or S3.</li>
 66  * <li>"sandbox" will let you access the MTurk sandbox, and the real S3.</li>
 67  * <li>"real" mode accesses the real MTurk and the real S3.</li>
 68  * </ul>
 69  */
 70 function setMode(mode) {
 71 	javaTurKit.setMode(mode)
 72 	mturk = new MTurk()
 73 	s3 = new S3()
 74 }
